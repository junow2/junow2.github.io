```C++
#include <bits/stdc++.h>
using namespace std; 
#define endl '\n'
typedef long long ll;
#define MAX 2000000
const int INF = 10002; 

int t, N, num[102];
ll ans;

int f(int x, int y) {

  return y? f(y, x%y) : x;
}

int main(void) {
  ios_base::sync_with_stdio(false); cin.tie(NULL);

  cin >> t;
  for (int i=0; i<t; i++) {
    cin >> N; ans = 0;
    for (int j=0; j<N; j++) cin >> num[j];

    sort(num, num+N); 
    // for (int j=0; j<N; j++) cout << num[j] << " "; cout << endl;

    for (int j=0; j<N; j++) 
      for (int k=j+1; k<N; k++) ans += f(num[j], num[k]); 

    cout << ans << endl;
  } 
}```
