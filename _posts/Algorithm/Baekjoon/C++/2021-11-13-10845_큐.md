```C++
#include <bits/stdc++.h>
using namespace std;
typedef long long li;

int main(void) {
  ios_base::sync_with_stdio(false); cin.tie(NULL);

  int N;
  queue <int> q;
  cin >> N;
  for (int i=0; i < N; i++) {
    string com;
    int num;
    cin >> com;

    if (com == "push") {
      cin >> num;
      q.push(num);
    }
    if (com == "pop") {
      if (q.empty()) {
        cout << -1 << '\n';
        continue;
      } 
      cout << q.front() << '\n';
      q.pop();
    }
    if (com == "size") cout << q.size() << '\n';
    if (com == "empty") {
      if (q.empty()) cout << 1 << '\n';
      else cout << 0 << '\n';
    }
    if (com == "front") {
      if (q.empty()) {
        cout << -1 << '\n';
        continue;
      }
      cout << q.front() << '\n';
    }
    if (com == "back") {
      if (q.empty()) {
        cout << -1 << '\n';
        continue;
      }
      cout << q.back() << '\n';
    }
  }  
}
```
